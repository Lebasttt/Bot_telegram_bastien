        """

            return None

        succes = [r for r in resultats_recentes if r["succes"]]
        echecs = [r for r in resultats_recentes if not r["succes"]]

        contexte_echecs = [r["contexte_systeme"] for r in echecs]

        if differences:
            correlation = CorrelationContexte(
                commande=commande,
                differences_contextuelles=differences,
                taux_succes=len(succes) / len(resultats_recentes),
                nombre_observations=len(resultats_recentes)
            )

        return None

    def analyser_dangers_et_blacklister(self, content):
        """Ajoute à la blacklist dynamique les commandes qui échouent ou sont dangereuses"""
        global DYNAMIC_BLACKLIST

        # Exemple: Blacklister les commandes qui échouent systématiquement
        echecs = re.findall(r"COMMANDE: (.*?)\n.*?CONCLUSION: ÉCHEC", content, re.DOTALL)
        for cmd in echecs:
            cmd_base = cmd.split()[0] + " " if cmd.split() else cmd
            if cmd_base not in DYNAMIC_BLACKLIST and cmd_base not in IMMUTABLE_BLACKLIST:
                DYNAMIC_BLACKLIST.append(cmd_base)
                log(f"🛡️ AUTO-PROTECTION: '{cmd_base}' ajouté à la blacklist (échec répété)")

    def analyser_echecs_pour_patterns(self, content):
        """Extrait des patterns qui mènent systématiquement à l'échec"""

        echecs_complets = re.findall(r"=== APPRENTISSAGE.*?COMMANDE: (.*?)\n.*?ERREUR_EXACTE: (.*?)\n.*?CONCLUSION: ÉCHEC", content, re.DOTALL)

        for commande, erreur in echecs_complets:
            # Simple extraction du premier mot de la commande
            mot_cle = commande.split()[0]
            if mot_cle not in self.patterns_mortels and mot_cle not in IMMUTABLE_BLACKLIST:
                # Logique simplifiée: si le mot-clé apparaît dans plus de 5 échecs, il devient un pattern mortel
                if content.count(f"COMMANDE: {mot_cle}") > 5 and content.count(f"CONCLUSION: ÉCHEC") > 5:
                    self.patterns_mortels.add(mot_cle)
                    log(f"🚫 PATTERN MORTEL IDENTIFIÉ: {mot_cle}")

def analyser_memoire_ram(self):
        """Analyse la mémoire RAM"""
        resultat = "🧠 MÉMOIRE RAM\n\n"
        mem_info = execute_commande_securisee("cat /proc/meminfo")
        if mem_info:
            resultat += f"Info mémoire:\n{mem_info}\n"

        # Détection problèmes mémoire
        if "MemAvailable" in mem_info:
            total, available = 0, 0
            for line in mem_info.split('\n'):
                if "MemTotal" in line:
                    total = int(''.join(filter(str.isdigit, line.split(':')[1])))
                elif "MemAvailable" in line:
                    available = int(''.join(filter(str.isdigit, line.split(':')[1])))

            if total > 0 and available > 0:
                pourcentage_utilise = ((total - available) / total) * 100
                if pourcentage_utilise >= self.alertes_seuil['memoire']:
                    self.problemes_critiques.append(f"📛 Mémoire RAM saturée (> {self.alertes_seuil['memoire']}%)")
                    self.suggestions_amélioration.append("🔧 Fermer les applications en arrière-plan")
        return resultat
